Examples - String.xsplit

Example Action Result
alert(jsdk.obj("1 2,3 5,6 7 8,0,10,11").xsplit("",",").join("|")); 1 2|3 5|6 7 8|0|10|11
alert(jsdk.obj("1 2,3 5,6 7 8,0,10,11").xsplit("first",[","," "]).join("|")); 1 2|3 5|6 7 8|0|10|11
alert(jsdk.obj("1 2,3 5,6 7 8,0,10,11").xsplit("first",[" ",","]).join("|")); 1|2,3|5,6|7|8,0,10,11
alert(jsdk.obj("1 2,3 5,6 7 8,0,10,11").xsplit("any",[","," "]).join("|")); 1|2|3|5|6|7|8|0|10|11
